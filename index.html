Conflict Resolution


@AvroDefault("\"unknown\"")
public final String user;

@Nullable
public final String origin;

@AvroIgnore
private boolean processed = false;

public Click(String time, String ip, String page, String user, String origin) {
    super(time, ip, page);
    this.user = user;
    this.origin = origin;
}
